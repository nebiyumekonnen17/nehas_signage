async function uploadAndDeploy() { const screenId = document.getElementById('screen-select').value; const fileInput = document.getElementById('media-file'); const statusMsg = document.getElementById('deploy-status'); const progressWrapper = document.getElementById('progress-wrapper'); const progressBar = document.getElementById('progress-bar'); const progressText = document.getElementById('progress-percent'); const deployBtn = document.getElementById('deploy-btn'); if (!screenId || !fileInput.files[0]) { statusMsg.textContent = "Please select a screen and file."; return; } const file = fileInput.files[0]; const fileExt = file.name.split('.').pop().toLowerCase(); const fileName = `${Math.random().toString(36).substring(2, 12)}_${Date.now()}.${fileExt}`; const filePath = `uploads/${fileName}`; deployBtn.disabled = true; progressWrapper.style.display = 'block'; progressBar.style.width = '0%'; progressText.textContent = '0%'; statusMsg.textContent = "Starting upload..."; try { // FIXED UPLOAD LOGIC const { data, error } = await supabaseClient.storage.from(BUCKET_NAME).upload(filePath, file, { cacheControl: '3600', upsert: false, duplex: 'half', // Required for some browsers to track progress correctly onUploadProgress: (progress) => { // Force calculation of percentage const percent = Math.round((progress.loaded / progress.total) * 100); // Update UI immediately progressBar.style.width = percent + '%'; progressText.textContent = percent + '%'; statusMsg.textContent = `Uploading: ${percent}% (${(progress.loaded / 1024 / 1024).toFixed(1)} MB sent)`; } }); if (error) throw error; statusMsg.textContent = "Syncing with screen..."; const { data: urlData } = supabaseClient.storage.from(BUCKET_NAME).getPublicUrl(filePath); const { data: mediaData } = await supabaseClient.from('media').insert([{ file_name: file.name, file_url: urlData.publicUrl, media_type: (fileExt === 'mp4' ? 'video' : 'image') }]).select(); const { data: currentItems } = await supabaseClient.from('playlist_items') .select('display_order').eq('screen_id', screenId) .order('display_order', { ascending: false }).limit(1); const nextOrder = (currentItems?.length ? currentItems[0].display_order : 0) + 1; await supabaseClient.from('playlist_items').insert([{ screen_id: screenId, media_id: mediaData[0].id, display_order: nextOrder, duration_seconds: document.getElementById('duration').value }]); statusMsg.style.color = '#10b981'; statusMsg.textContent = "Success! Content is live."; } catch (err) { console.error("Upload error:", err); statusMsg.style.color = '#dc2626'; statusMsg.textContent = "Upload failed. If it's a large video, check Supabase limits."; } finally { deployBtn.disabled = false; // Keep the progress bar visible for 2 seconds so they can see it hit 100% setTimeout(() => { progressWrapper.style.display = 'none'; document.getElementById('file-details-box').style.display = 'none'; fileInput.value = ''; }, 2000); } }
